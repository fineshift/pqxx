955 class PQXX_LIBEXPORT scoped_noticer : errorhandler
956 {
957 public:

959  errorhandler(c), m_noticer(t.release()) {}

961  scoped_noticer(connection_base &c, noticer *t) throw () :
962  errorhandler(c), m_noticer(t) {}
